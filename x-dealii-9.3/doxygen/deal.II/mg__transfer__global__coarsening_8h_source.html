577  initialize_dof_vector(level, dst[level]);
578 
579  dst[transfer.max_level()].copy_locally_owned_data_from(src);
580 
581  for (unsigned int l = max_level; l > min_level; --l)
582  this->transfer[l].interpolate(dst[l - 1], dst[l]);
