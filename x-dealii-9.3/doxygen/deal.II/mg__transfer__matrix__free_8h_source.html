505 template <int dim, typename Number>
506 template <typename Number2, int spacedim>
507 void
508 MGTransferMatrixFree<dim, Number>::interpolate_to_mg(
509  const DoFHandler<dim, spacedim> & dof_handler,
510  MGLevelObject<LinearAlgebra::distributed::Vector<Number>> &dst,
511  const LinearAlgebra::distributed::Vector<Number2> & src) const
512 {
513  const unsigned int min_level = dst.min_level();
514  const unsigned int max_level = dst.max_level();
515 
516  Assert(max_level == dof_handler.get_triangulation().n_global_levels() - 1,
517  ExcDimensionMismatch(
518  max_level, dof_handler.get_triangulation().n_global_levels() - 1));
519 
520  const FiniteElement<dim, spacedim> &fe = dof_handler.get_fe();
521 
522  for (unsigned int level = min_level; level <= max_level; ++level)
523  if (dst[level].size() != dof_handler.n_dofs(level) ||
524  dst[level].locally_owned_size() !=
525  dof_handler.locally_owned_mg_dofs(level).n_elements())
526  dst[level].reinit(this->vector_partitioners[level]);
527 
528  // copy fine level vector to active cells in MG hierarchy
529  this->copy_to_mg(dof_handler, dst, src, true);
530 
531  // FIXME: maybe need to store hanging nodes constraints per level?
532  // MGConstrainedDoFs does NOT keep this info right now, only periodicity
533  // constraints...
534 
535  // do the transfer from level to level-1:
536  dst[max_level].update_ghost_values();
537  for (unsigned int level = max_level; level > min_level; --level)
538  {
539  // auxiliary vector which always has ghost elements
540  const LinearAlgebra::distributed::Vector<Number> *input = nullptr;
541  LinearAlgebra::distributed::Vector<Number> ghosted_fine;
542  if (dst[level].get_partitioner().get() ==
543  this->vector_partitioners[level].get())
544  input = &dst[level];
545  else
546  {
547  ghosted_fine.reinit(this->vector_partitioners[level]);
548  ghosted_fine.copy_locally_owned_data_from(dst[level]);
549  ghosted_fine.update_ghost_values();
550  input = &ghosted_fine;
551  }
552 
553  std::vector<Number> dof_values_coarse(fe.n_dofs_per_cell());
554  Vector<Number> dof_values_fine(fe.n_dofs_per_cell());
555  Vector<Number> tmp(fe.n_dofs_per_cell());
556  std::vector<types::global_dof_index> dof_indices(fe.n_dofs_per_cell());
557  for (const auto &cell : dof_handler.cell_iterators_on_level(level - 1))
558  if (cell->is_locally_owned_on_level())
559  {
560  // if we get to a cell without children (== active), we can
561  // skip it as there values should be already set by the
562  // equivalent of copy_to_mg()
563  if (cell->is_active())
564  continue;
565 
566  std::fill(dof_values_coarse.begin(), dof_values_coarse.end(), 0.);
567  for (unsigned int child = 0; child < cell->n_children(); ++child)
568  {
569  cell->child(child)->get_mg_dof_indices(dof_indices);
570  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
571  dof_values_fine(i) = (*input)(dof_indices[i]);
572  fe.get_restriction_matrix(child, cell->refinement_case())
573  .vmult(tmp, dof_values_fine);
574  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
575  if (fe.restriction_is_additive(i))
576  dof_values_coarse[i] += tmp[i];
577  else if (tmp(i) != 0.)
578  dof_values_coarse[i] = tmp[i];
579  }
580  cell->get_mg_dof_indices(dof_indices);
581  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
582  if (dof_handler.locally_owned_mg_dofs(level - 1).is_element(
583  dof_indices[i]))
584  dst[level - 1](dof_indices[i]) = dof_values_coarse[i];
585  }
586 
587  dst[level - 1].update_ghost_values();
588  }
589 }

615 template <int dim, typename Number>
616 template <typename Number2, int spacedim>
617 void
618 MGTransferBlockMatrixFree<dim, Number>::copy_to_mg(
619  const std::vector<const DoFHandler<dim, spacedim> *> & dof_handler,
620  MGLevelObject<LinearAlgebra::distributed::BlockVector<Number>> &dst,
621  const LinearAlgebra::distributed::BlockVector<Number2> & src) const
622 {
623  const unsigned int n_blocks = src.n_blocks();
624  AssertDimension(dof_handler.size(), n_blocks);
625 
626  if (n_blocks == 0)
627  return;
628 
629  const unsigned int min_level = dst.min_level();
630  const unsigned int max_level = dst.max_level();
631 
632  // this function is normally called within the Multigrid class with
633  // dst == defect level block vector. At first run this vector is not
634  // initialized. Do this below:
635  {
636  const parallel::TriangulationBase<dim, spacedim> *tria =
637  (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
638  &(dof_handler[0]->get_triangulation())));
639  for (unsigned int i = 1; i < n_blocks; ++i)
640  AssertThrow(
641  (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
642  &(dof_handler[0]->get_triangulation())) == tria),
643  ExcMessage("The DoFHandler use different Triangulations!"));
644 
645  MGLevelObject<bool> do_reinit;
646  do_reinit.resize(min_level, max_level);
647  for (unsigned int level = min_level; level <= max_level; ++level)
648  {
649  do_reinit[level] = false;
650  if (dst[level].n_blocks() != n_blocks)
651  {
652  do_reinit[level] = true;
653  continue; // level
654  }
655  for (unsigned int b = 0; b < n_blocks; ++b)
656  {
657  LinearAlgebra::distributed::Vector<Number> &v = dst[level].block(b);
658  if (v.size() !=
659  dof_handler[b]->locally_owned_mg_dofs(level).size() ||
660  v.locally_owned_size() !=
661  dof_handler[b]->locally_owned_mg_dofs(level).n_elements())
662  {
663  do_reinit[level] = true;
664  break; // b
665  }
666  }
667  }
668 
669  for (unsigned int level = min_level; level <= max_level; ++level)
670  {
671  if (do_reinit[level])
672  {
673  dst[level].reinit(n_blocks);
674  for (unsigned int b = 0; b < n_blocks; ++b)
675  {
676  LinearAlgebra::distributed::Vector<Number> &v =
677  dst[level].block(b);
678  v.reinit(dof_handler[b]->locally_owned_mg_dofs(level),
679  dof_handler[b]->get_communicator());
680  }
681  dst[level].collect_sizes();
682  }
683  else
684  dst[level] = 0;
685  }
686  }
687 
688  // FIXME: this a quite ugly as we need a temporary object:
689  MGLevelObject<LinearAlgebra::distributed::Vector<Number>> dst_non_block(
690  min_level, max_level);
691 
692  for (unsigned int b = 0; b < n_blocks; ++b)
693  {
694  for (unsigned int l = min_level; l <= max_level; ++l)
695  dst_non_block[l].reinit(dst[l].block(b));
696  const unsigned int data_block = same_for_all ? 0 : b;
697  matrix_free_transfer_vector[data_block].copy_to_mg(*dof_handler[b],
698  dst_non_block,
699  src.block(b));
700 
701  for (unsigned int l = min_level; l <= max_level; ++l)
702  dst[l].block(b) = dst_non_block[l];
703  }
704 }
