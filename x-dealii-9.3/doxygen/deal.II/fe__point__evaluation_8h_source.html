549 template <int n_components, int dim, int spacedim>
550 FEPointEvaluation<n_components, dim, spacedim>::FEPointEvaluation(
551  const Mapping<dim> & mapping,
552  const FiniteElement<dim> &fe,
553  const unsigned int first_selected_component)
554  : mapping(&mapping)
555  , mapping_q_generic(
556  dynamic_cast<const MappingQGeneric<dim, spacedim> *>(&mapping))
557  , fe(&fe)
558 {
559  if (mapping_q_generic != nullptr &&
560  internal::MatrixFreeFunctions::ShapeInfo<double>::is_supported(fe))
561  {
562  internal::MatrixFreeFunctions::ShapeInfo<double> shape_info;
563  unsigned int base_element_number = 0;
564  unsigned int component = 0;
565  for (; base_element_number < fe.n_base_elements(); ++base_element_number)
566  if (component + fe.element_multiplicity(base_element_number) >
567  first_selected_component)
568  {
569  Assert(first_selected_component + n_components <=
570  component + fe.element_multiplicity(base_element_number),
571  ExcMessage("You selected an evaluation which crosses "
572  "different base elements, a case not supported"));
573  break;
574  }
575  else
576  component += fe.element_multiplicity(base_element_number);
577 
578  shape_info.reinit(QMidpoint<1>(), fe, base_element_number);
579  renumber = shape_info.lexicographic_numbering;
580  dofs_per_component = shape_info.dofs_per_component_on_cell;
581  poly = Polynomials::generate_complete_Lagrange_basis(
582  QGaussLobatto<1>(shape_info.data[0].fe_degree + 1).get_points());
583  }
584  if (true /*TODO: as long as the fast path of integrate() is not working*/)
585  {
586  nonzero_shape_function_component.resize(fe.n_dofs_per_cell());
587  for (unsigned int d = 0; d < n_components; ++d)
588  {
589  const unsigned int component = first_selected_component + d;
590  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
591  {
592  const bool is_primitive = fe.is_primitive() || fe.is_primitive(i);
593  if (is_primitive)
594  nonzero_shape_function_component[i][d] =
595  (component == fe.system_to_component_index(i).first);
596  else
597  nonzero_shape_function_component[i][d] =
598  (fe.get_nonzero_components(i)[component] == true);
599  }
600  }
601  }
602 }

606 template <int n_components, int dim, int spacedim>
607 void
608 FEPointEvaluation<n_components, dim, spacedim>::evaluate(
609  const typename Triangulation<dim, spacedim>::cell_iterator &cell,
610  const ArrayView<const Point<dim>> & unit_points,
611  const ArrayView<const double> & solution_values,
612  const EvaluationFlags::EvaluationFlags & evaluation_flag)
613 {
614  AssertDimension(solution_values.size(), fe->dofs_per_cell);
615  if (((evaluation_flag & EvaluationFlags::values) ||
616  (evaluation_flag & EvaluationFlags::gradients)) &&
617  !poly.empty())
618  {
619  // fast path with tensor product evaluation
620  if (solution_renumbered.size() != dofs_per_component)
621  solution_renumbered.resize(dofs_per_component);
622  for (unsigned int comp = 0; comp < n_components; ++comp)
623  for (unsigned int i = 0; i < dofs_per_component; ++i)
624  internal::FEPointEvaluation::EvaluatorTypeTraits<dim, n_components>::
625  read_value(solution_values[renumber[comp * dofs_per_component + i]],
626  comp,
627  solution_renumbered[i]);
628 
629  if (evaluation_flag & EvaluationFlags::values)
630  values.resize(unit_points.size());
631  if (evaluation_flag & EvaluationFlags::gradients)
632  gradients.resize(unit_points.size());
633 
634  const std::size_t n_points = unit_points.size();
635  const std::size_t n_lanes = VectorizedArray<double>::size();
636  for (unsigned int i = 0; i < n_points; i += n_lanes)
637  {
638  // convert to vectorized format
639  Point<dim, VectorizedArray<double>> vectorized_points;
640  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
641  for (unsigned int d = 0; d < dim; ++d)
642  vectorized_points[d][j] = unit_points[i + j][d];
643 
644  // compute
645  const auto val_and_grad =
646  internal::evaluate_tensor_product_value_and_gradient(
647  poly, solution_renumbered, vectorized_points, poly.size() == 2);
648 
649  // convert back to standard format
650  if (evaluation_flag & EvaluationFlags::values)
651  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
652  internal::FEPointEvaluation::EvaluatorTypeTraits<
653  dim,
654  n_components>::set_value(val_and_grad.first, j, values[i + j]);
655  if (evaluation_flag & EvaluationFlags::gradients)
656  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
657  internal::FEPointEvaluation::
658  EvaluatorTypeTraits<dim, n_components>::set_gradient(
659  val_and_grad.second, j, gradients[i + j]);
660  }
661 
662  // let mapping compute the transformation
663  if (evaluation_flag & EvaluationFlags::gradients)
664  {
665  Assert(mapping_q_generic != nullptr, ExcInternalError());
666  mapping_q_generic->transform_variable(
667  cell,
668  mapping_covariant,
669  unit_points,
670  ArrayView<const gradient_type>(gradients.data(), gradients.size()),
671  ArrayView<gradient_type>(gradients.data(), gradients.size()));
672  }
673  }
674  else if ((evaluation_flag & EvaluationFlags::values) ||
675  (evaluation_flag & EvaluationFlags::gradients))
676  {
677  // slow path with FEValues
678  const UpdateFlags flags =
679  ((evaluation_flag & EvaluationFlags::values) ? update_values :
680  update_default) |
681  ((evaluation_flag & EvaluationFlags::gradients) ? update_gradients :
682  update_default);
683  FEValues<dim, spacedim> fe_values(
684  *mapping,
685  *fe,
686  Quadrature<dim>(
687  std::vector<Point<dim>>(unit_points.begin(), unit_points.end())),
688  flags);
689  fe_values.reinit(cell);
690 
691  if (evaluation_flag & EvaluationFlags::values)
692  {
693  values.resize(unit_points.size());
694  std::fill(values.begin(), values.end(), value_type());
695  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
696  {
697  const double value = solution_values[i];
698  for (unsigned int d = 0; d < n_components; ++d)
699  if (nonzero_shape_function_component[i][d] &&
700  (fe->is_primitive(i) || fe->is_primitive()))
701  for (unsigned int q = 0; q < unit_points.size(); ++q)
702  internal::FEPointEvaluation::
703  EvaluatorTypeTraits<dim, n_components>::access(values[q],
704  d) +=
705  fe_values.shape_value(i, q) * value;
706  else if (nonzero_shape_function_component[i][d])
707  for (unsigned int q = 0; q < unit_points.size(); ++q)
708  internal::FEPointEvaluation::
709  EvaluatorTypeTraits<dim, n_components>::access(values[q],
710  d) +=
711  fe_values.shape_value_component(i, q, d) * value;
712  }
713  }
714 
715  if (evaluation_flag & EvaluationFlags::gradients)
716  {
717  gradients.resize(unit_points.size());
718  std::fill(gradients.begin(), gradients.end(), gradient_type());
719  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
720  {
721  const double value = solution_values[i];
722  for (unsigned int d = 0; d < n_components; ++d)
723  if (nonzero_shape_function_component[i][d] &&
724  (fe->is_primitive(i) || fe->is_primitive()))
725  for (unsigned int q = 0; q < unit_points.size(); ++q)
726  internal::FEPointEvaluation::
727  EvaluatorTypeTraits<dim, n_components>::access(
728  gradients[q], d) += fe_values.shape_grad(i, q) * value;
729  else if (nonzero_shape_function_component[i][d])
730  for (unsigned int q = 0; q < unit_points.size(); ++q)
731  internal::FEPointEvaluation::EvaluatorTypeTraits<
732  dim,
733  n_components>::access(gradients[q], d) +=
734  fe_values.shape_grad_component(i, q, d) * value;
735  }
736  }
737  }
738 }
