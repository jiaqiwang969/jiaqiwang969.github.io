1099  template <class VectorType>
1100  inline typename std::enable_if<
1101  std::is_convertible<VectorType *, Function<spacedim> *>::value ==
1102  false>::type
1103  ParticleHandler<dim, spacedim>::set_particle_positions(
1104  const VectorType &input_vector,
1105  const bool displace_particles)
1106  {
1107  AssertDimension(input_vector.size(),
1108  get_next_free_particle_index() * spacedim);
1109  for (auto &p : *this)
1110  {
1111  auto new_point(displace_particles ? p.get_location() :
1112  Point<spacedim>());
1113  const auto id = p.get_id();
1114  for (unsigned int i = 0; i < spacedim; ++i)
1115  new_point[i] += input_vector[id * spacedim + i];
1116  p.set_location(new_point);
1117  }
1118  sort_particles_into_subdomains_and_cells();
1119  }

1126  ParticleHandler<dim, spacedim>::get_particle_positions(

1137  for (unsigned int i = 0; i < spacedim; ++i)

1139  else

1141  output_vector[id * spacedim + i] = point[i];
1142  }
