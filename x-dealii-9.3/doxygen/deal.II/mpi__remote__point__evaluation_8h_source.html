263  ArrayView<T> buffer_2(buffer.data() + buffer.size() / 2,
264  buffer.size() / 2);
265 
266  // evaluate functions at points
267  evaluation_function(buffer_1, cell_data);
268 
269  // sort for communication
270  for (unsigned int i = 0; i < send_permutation.size(); ++i)
271  buffer_2[send_permutation[i]] = buffer_1[i];
272 
273  // process remote quadrature points and send them away
274  std::map<unsigned int, std::vector<char>> temp_map;
275 
276  std::vector<MPI_Request> requests;
277  requests.reserve(send_ranks.size());
278 
279  const unsigned int my_rank =
280  Utilities::MPI::this_mpi_process(tria->get_communicator());
281 
282  std::map<unsigned int, std::vector<T>> temp_recv_map;
283 
284  for (unsigned int i = 0; i < send_ranks.size(); ++i)
285  {
286  if (send_ranks[i] == my_rank)
287  {
288  // process locally-owned values
289  temp_recv_map[my_rank] =
290  std::vector<T>(buffer_2.begin() + send_ptrs[i],
291  buffer_2.begin() + send_ptrs[i + 1]);
292  continue;
293  }
294 
295  temp_map[send_ranks[i]] =
296  Utilities::pack(std::vector<T>(buffer_2.begin() + send_ptrs[i],
297  buffer_2.begin() + send_ptrs[i + 1]),
298  false);
299 
300  auto &buffer = temp_map[send_ranks[i]];
301 
302  requests.push_back(MPI_Request());
303 
304  MPI_Isend(buffer.data(),
305  buffer.size(),

308  internal::Tags::remote_point_evaluation,
309  tria->get_communicator(),

311  }

314  {
315  if (recv_rank == my_rank)
316  continue;
317

320  internal::Tags::remote_point_evaluation,
321  tria->get_communicator(),
322  &status);
323 
324  int message_length;
325  MPI_Get_count(&status, MPI_CHAR, &message_length);
326 
327  std::vector<char> buffer(message_length);
328 
329  MPI_Recv(buffer.data(),
330  buffer.size(),
331  MPI_CHAR,
332  status.MPI_SOURCE,
333  internal::Tags::remote_point_evaluation,
334  tria->get_communicator(),
335  MPI_STATUS_IGNORE);
336

338  Utilities::unpack<std::vector<T>>(buffer, false);
339  }
340

342  MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
343

345  auto it = recv_permutation.begin();
346  for (const auto &j : temp_recv_map)
347  for (const auto &i : j.second)
