57  void
58  RemotePointEvaluation<dim, spacedim>::reinit(
59  const std::vector<Point<spacedim>> &points,
60  const Triangulation<dim, spacedim> &tria,
61  const Mapping<dim, spacedim> & mapping)
62  {
63 #ifndef DEAL_II_WITH_MPI
64  Assert(false, ExcNeedsMPI());
65  (void)points;
66  (void)tria;
67  (void)mapping;
68 #else
69  if (tria_signal.connected())
70  tria_signal.disconnect();
71 
72  tria_signal =
73  tria.signals.any_change.connect([&]() { this->ready_flag = false; });
74 
75  this->tria = &tria;
76  this->mapping = &mapping;
77 
78  std::vector<BoundingBox<spacedim>> local_boxes;
79  for (const auto &cell : tria.active_cell_iterators())
80  if (cell->is_locally_owned())
81  local_boxes.push_back(mapping.get_bounding_box(cell));
82 
83  // create r-tree of bounding boxes
84  const auto local_tree = pack_rtree(local_boxes);
85 
86  // compress r-tree to a minimal set of bounding boxes
87  const auto local_reduced_box = extract_rtree_level(local_tree, 0);
88 
89  // gather bounding boxes of other processes
90  const auto global_bboxes =
91  Utilities::MPI::all_gather(tria.get_communicator(), local_reduced_box);
92 
93  const GridTools::Cache<dim, spacedim> cache(tria, mapping);
94 
95  const auto data =
96  GridTools::internal::distributed_compute_point_locations(
97  cache, points, global_bboxes, tolerance, true);
98 
99  this->recv_ranks = data.recv_ranks;
100  this->recv_ptrs = data.recv_ptrs;
101 
102  this->send_ranks = data.send_ranks;
103  this->send_ptrs = data.send_ptrs;
104 
105  this->recv_permutation = {};
106  this->recv_permutation.resize(data.recv_components.size());
107  this->point_ptrs.assign(points.size() + 1, 0);
108  for (unsigned int i = 0; i < data.recv_components.size(); ++i)
109  {
110  AssertIndexRange(std::get<2>(data.recv_components[i]),
111  this->recv_permutation.size());
112  this->recv_permutation[std::get<2>(data.recv_components[i])] = i;
113 
114  AssertIndexRange(std::get<1>(data.recv_components[i]) + 1,
115  this->point_ptrs.size());
116  this->point_ptrs[std::get<1>(data.recv_components[i]) + 1]++;
117  }
118 
119  unique_mapping = true;
120  for (unsigned int i = 0; i < points.size(); ++i)
121  {
122  if (unique_mapping && this->point_ptrs[i + 1] != 1)
123  unique_mapping = false;
124  this->point_ptrs[i + 1] += this->point_ptrs[i];
125  }
126 
127  cell_data = {};
128  send_permutation = {};
129 
130  std::pair<int, int> dummy{-1, -1};
131  for (const auto &i : data.send_components)
132  {
133  if (dummy != std::get<0>(i))
134  {
135  dummy = std::get<0>(i);
136  cell_data.cells.emplace_back(dummy);
137  cell_data.reference_point_ptrs.emplace_back(
138  cell_data.reference_point_values.size());
139  }
140 
141  cell_data.reference_point_values.emplace_back(std::get<3>(i));
142  send_permutation.emplace_back(std::get<5>(i));
143  }
144 
145  cell_data.reference_point_ptrs.emplace_back(
146  cell_data.reference_point_values.size());
147 
148  this->ready_flag = true;
149 #endif
150  }
